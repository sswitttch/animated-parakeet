# animated-turtle

import turtle
 
turtle.speed(10)
for i in range(200):
    turtle.forward(i)
    turtle.right(98)
    turtle.pencolor('red')
    turtle.pencolor('orange')
    turtle.pencolor('yellow')
    turtle.pencolor('green')
    turtle.pencolor('cyan')
    turtle.pencolor('blue')
    turtle.pencolor('purple')
 
turtle.done()
